# ─── AWS SSO profile switcher ──────────────────────────────────────────
# Usage:  awsp <profile>        e.g.  awsp synapse-admin
#         awsp                  (lists available profiles)
#
# - Accepts short names: "synapse-admin" resolves to "payfacto-synapse-admin"
# - Only triggers an SSO browser login if the current token is missing/expired
# - Exports AWS_PROFILE into the *current* shell (that's why this is a function)
# - Prints the resolved Account + ARN so you can confirm who you are
awsp() {
  local input="$1" profile=""

  if [ -z "$input" ]; then
    echo "Usage: awsp <profile>"
    echo "Profiles:"
    aws configure list-profiles 2>/dev/null | sed 's/^/  /'
    return 1
  fi

  # Resolve name: exact match first, then try the payfacto- prefix
  if aws configure list-profiles 2>/dev/null | grep -qx "$input"; then
    profile="$input"
  elif aws configure list-profiles 2>/dev/null | grep -qx "payfacto-$input"; then
    profile="payfacto-$input"
  else
    echo "Unknown profile: '$input'" >&2
    echo "Profiles:" >&2
    aws configure list-profiles 2>/dev/null | sed 's/^/  /' >&2
    return 1
  fi

  # Reuse the existing SSO session if it's still valid; only log in if not
  if ! aws sts get-caller-identity --profile "$profile" >/dev/null 2>&1; then
    echo "Logging in to SSO for $profile ..."
    aws sso login --profile "$profile" || return 1
  fi

  export AWS_PROFILE="$profile"
  echo "AWS_PROFILE=$profile"
  aws sts get-caller-identity --query '[Account,Arn]' --output text
}

# Tab-completion for awsp (full names and short, payfacto-stripped names)
_awsp() {
  local profiles short
  profiles="$(aws configure list-profiles 2>/dev/null)"
  short="$(echo "$profiles" | sed 's/^payfacto-//')"
  COMPREPLY=( $(compgen -W "$profiles $short" -- "${COMP_WORDS[COMP_CWORD]}") )
}
complete -F _awsp awsp

# awswho: print the current identity without switching profiles
awswho() {
  echo "AWS_PROFILE=${AWS_PROFILE:-(unset -> uses [default])}"
  if ! aws sts get-caller-identity --query '[Account,Arn]' --output text 2>/dev/null; then
    echo "Not authenticated for this profile (run: awsp <profile>)" >&2
    return 1
  fi
}
# ─── end AWS SSO profile switcher ──────────────────────────────────────
